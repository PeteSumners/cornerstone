## Web Audio API Implementation

See `schumann-audio-generator.js` for full implementation.

### Basic Example

```javascript
const audioContext = new AudioContext();

// Create binaural beat at 7.83 Hz
function createSchumannBinauralBeat(baseFreq = 200) {
  // Left channel
  const leftOsc = audioContext.createOscillator();
  leftOsc.frequency.value = baseFreq;

  // Right channel (7.83 Hz higher)
  const rightOsc = audioContext.createOscillator();
  rightOsc.frequency.value = baseFreq + 7.83;

  // Pan to stereo
  const leftPanner = audioContext.createStereoPanner();
  leftPanner.pan.value = -1; // Full left

  const rightPanner = audioContext.createStereoPanner();
  rightPanner.pan.value = 1; // Full right

  // Connect and play
  leftOsc.connect(leftPanner).connect(audioContext.destination);
  rightOsc.connect(rightPanner).connect(audioContext.destination);

  leftOsc.start();
  rightOsc.start();

  return { leftOsc, rightOsc };
}
```
